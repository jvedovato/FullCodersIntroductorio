Algoritmo gastoPromedio
	gastoAcumulado<-0
	Escribir "¿Cuántos meses va a evaluar?";
	Leer cantidadMeses
	para x<-1 hasta cantidadMeses
		Escribir "Ingrese el gasto del mes ", x

	FinPara
	promedio<-gastoAcumulado/cantidadMeses;
